208static inline void
209model_cone_vertex (const float r, const float height, const float theta, const float h, unsigned int *pos,
210 float *points)
211{
212 /*
213 x = r ⋅ (h/height) ⋅ cos(θ)
214 y = 2 ⋅ (height + h)
215 z = r ⋅ (h/height) ⋅ sin(θ)
216
217 r ≥ 0
218 θ ∈ {-π + i⋅s : s = 2π/slices ∧ i ∈ {0,...,slices} }
219 h ∈ {-height + j⋅t : t = height/stacks ∧ j ∈ {0,...,stacks} }
220
221 check:
222 1. https://www.math3d.org/7oeSkmuns
223 */
224 points_vertex (r * h / height * cos (theta), height + h, r * h / height * sin (theta), pos, points);
225}

347vector<array<vec3, 16>> read_Bezier (string &patch)
348{
349 string buffer;
350 ifstream myFile;
351
352 myFile.open (patch, ios::in | ios::out);
353 getline (myFile, buffer);
354 // Número de patches presentes no ficheiro.
355 int n_patches = stoi (buffer);
356
357 // Vetor de vetores de índices.
358 vector<vector<int>> patches;
359
360 // Ciclo externo lê uma linha (patch) de cada vez
361 for (int j = 0; j < n_patches; j++)
362 {
363 vector<int> patchIndexes;
364 /*
365 Ciclo interno lê os índices dos pontos de controlo de cada patch, sabendo que cada
366 patch terá 16 pontos de controlo.
367 */
368 for (int i = 0; i < 16; i++)
369 {
370 getline (myFile, buffer, ',');
371 patchIndexes.push_back (stoi (buffer));
372 }
373 patches.push_back (patchIndexes);
374 }
375
376 getline (myFile, buffer);
377 // Número de pontos presentes no ficheiro.
378 int pts = stoi (buffer);
379
380 // Vetor que guardará as coordenadas de pontos de controlo para superfície de Bézier.
381 vector<vec3> control;
382 for (int j = 0; j < pts; j++)
383 {
384 vec3 v;
385 getline (myFile, buffer, ',');
386 v.x = stof (buffer);
387 getline (myFile, buffer, ',');
388 v.y = stof (buffer);
389 getline (myFile, buffer, ',');
390 v.z = stof (buffer);
391 control.push_back (v);
392 }
393
394 /*
395 Percorrem-se os vetores que, para cada patch, guardam os seus índices de pontos de controlo.
396 Para cada patch, constroi-se um vetor com as coordenadas dos seus pontos de controlo.
397 */
398 vector<array<vec3, 16>> pointsInPatches;
399 for (auto &patche : patches)
400 {
401 array<vec3, 16> pointsInPatch{};
402 for (int j = 0; j < 16; j++)
403 {
404 pointsInPatch[j] = control[patche[j]];
405 }
406 pointsInPatches.push_back (pointsInPatch);
407 }
408
409 myFile.close ();
410 return pointsInPatches;
411}

425template<typename T1, typename T2, typename T3>
426void mult (const T1 &a, const T2 &b, T3 &r, const int m, const int p, const int q)
427{
428 for (auto i = 0; i < m; ++i)
429 {
430 for (auto j = 0; j < q; ++j)
431 {
432 r[m * i + j] *= 0;
433 for (auto k = 0; k < p; k++)
434 {
435 r[m * i + j] += a[m * i + k] * b[p * k + j];
436 }
437 }
438 }

441vec3 get_bezier_point (const float u, const float v, const array<vec3, 16> &P)
442{
443 // B(u,v) = U M P Mᵀ V
444 vec4 U = dec_polynomial (u);
445 vec4 V = dec_polynomial (v);
446
447 vector<vec3> VMP;
448 auto VM = V * glm::transpose (Mb); //1x4x1
449 mult (VM, P, VMP, 1, 4, 4);
450
451 vector<vec3> VMPMU;
452 auto MU = Mb * U;
453 mult (VMP, MU, VMPMU, 1, 4, 1);
454 return VMPMU[0];
455}

457vector<vec3> get_bezier_surface (array<vec3, 16> control_points, int int_tesselation)
458{
459 vector<vec3> pontos;
460 float float_tesselation = (float) int_tesselation;
461 for (int tv = 0; tv < int_tesselation; ++tv)
462 {
463 float v = (float) tv / float_tesselation;
464 for (int tu = 0; tu < int_tesselation; ++tu)
465 {
466 float u = (float) tu / (float) float_tesselation;
467 // triângulo superior
468 pontos.push_back (get_bezier_point ((u + (1.0f / float_tesselation)), (v + (1.0f / float_tesselation)), control_points));
469 pontos.push_back (get_bezier_point (u, (v + (1.0f / float_tesselation)), control_points));
470 pontos.push_back (get_bezier_point (u, v, control_points));
471 // triângulo inferior
472 pontos.push_back (get_bezier_point (u, v, control_points));
473 pontos.push_back (get_bezier_point ((u + (1.0f / float_tesselation)), v, control_points));
474 pontos.push_back (get_bezier_point ((u + (1.0f / float_tesselation)), (v + (1.0f / float_tesselation)), control_points));
475 }
476 }
477 return pontos;
478}

75 const unsigned int divisions,
76 vector<vec3> &vertices,
77 vector<vec3> &normals,
78 vector<vec2> &texture)
79{
80 const float o = -length / 2.0f;
81 const float d = length / (float) divisions;
82
83 for (unsigned int uidiv1 = 1; uidiv1 <= divisions; ++uidiv1)
84 {
85 for (unsigned int uidiv2 = 1; uidiv2 <= divisions; ++uidiv2)
86 {
87 auto const fdiv1 = (float) uidiv1;
88 auto const fdiv2 = (float) uidiv2;
89 auto const fdivisions = (float) divisions;
90
91 for (auto e : {
92 array<float, 2>{-1, -1},//P1
93 array<float, 2>{-1, 0},//P1'z
94 array<float, 2>{0, -1},//P1'x
95

99 {
100 vertices.emplace_back (o + d * (fdiv1 + e[0]), 0, o + d * (fdiv2 + e[1]));
101 normals.emplace_back (0, 1, 0);
102 texture.emplace_back ((fdiv1 + e[0]) / fdivisions, (fdiv2 + e[1]) / fdivisions);

336 requires arithmetic<T>
337static inline void
338model_cone_vertex (const T r,
339 const T height,
340 const T theta,
341 const T h,
342 vector<vec3> &vertices)
343{
344 /*
345 x = r ⋅ (h/height) ⋅ cos(θ)
346 y = 2 ⋅ (height + h)
347 z = r ⋅ (h/height) ⋅ sin(θ)
348
349 r ≥ 0
350 θ ∈ {-π + i⋅s : s = 2π/slices ∧ i ∈ {0,...,slices} }
351 h ∈ {-height + j⋅t : t = height/stacks ∧ j ∈ {0,...,stacks} }
352
353 check:
354 1. https://www.math3d.org/7oeSkmuns
355 */
356 vertices.emplace_back (r * h / height * cos (theta), height + h, r * h / height * sin (theta));
357}

359template<typename T>
360 requires arithmetic<T>
361void model_cone_vertices (const T radius,
362 const T height,
363 const unsigned int slices,
364 const unsigned int stacks,
365 vector<vec3> &vertices,
366 vector<vec3> &normals,
367 vector<vec2> &texture)
368{
369
370 const T s = 2 * M_PI / (float) slices;
371 const T t = height / (float) stacks;
372
373 const T theta_0 = -M_PI;
374 const T h_0 = -height;
375
376 auto const fslices = (float) slices;
377 auto const fstacks = (float) stacks;
378
379 for (unsigned int slice = 1; slice <= slices; ++slice)
380 {
381 for (unsigned int stack = 1; stack <= stacks; ++stack)
382 {
383 auto const fslice = (float) slice;
384 auto const fstack = (float) stack;
385
386 //base
387 vertices.emplace_back (0, 0, 0); //O
388 texture.emplace_back (0, 0);
389 normals.emplace_back (0, -1, 0);
390 for (auto e : {
391 -1.0f,//P1
392 .0f //P2
393 })
394 {
395 model_cone_vertex (radius, height, theta_0 + s * (fslice + e), h_0, vertices); //P1
396 normals.emplace_back (0, -1, 0);
397 }
398
399 texture.emplace_back (-1, 0);
400 texture.emplace_back (0, 0);
401
402 int q = 0;
403 for (auto e : {
404 array<float, 2>{0, -1},
405 array<float, 2>{-1, 0},
406 array<float, 2>{0, 0},
407
408 array<float, 2>{0, -1},
409 array<float, 2>{-1, -1},
410 array<float, 2>{-1, 0},
411 })
412 {
413 model_cone_vertex (radius, height,
414 theta_0 + s * (fslice + e[0]),
415 h_0 + t * (fstack + e[1]), vertices);
416 if (q % 3 == 2)
417 {
418 const auto P1 = vertices.end ()[-2];
419 const auto P2 = vertices.end ()[-3];
420 const auto P1_prime = vertices.end ()[-1];
421 for (auto _ = 0; _ < 3; ++_)
422 normals.emplace_back (normalize (cross (P2 - P1_prime, P1 - P1_prime)));
423 }
424
425 texture.emplace_back (fslice / fslices, fstack / fstacks);
426 ++q;
427 }
428 }
429 }
430}

464static inline void
465model_sphere_vertex (const auto r,
466 const auto theta,
467 const auto phi,
468 vector<vec3> &vertices,
469 vector<vec3> &normals)
470{
471 /*
472 x = r ⋅ sin(θ)cos(φ)
473 y = r ⋅ sin(φ)
474 z = r ⋅ cos(θ)cos(φ)
475
476 r ≥ 0
477 θ ∈ {-π + i⋅s : s = 2π/slices ∧ i ∈ {0,...,slices} }
478 ϕ ∈ {-π/2 + j⋅t : t = π/stacks ∧ j ∈ {0,...,stacks} }
479
480 check
481 1. https://www.math3d.org/EumEEZBKe
482 2. https://www.math3d.org/zE4n6xayX
483 */
484 vertices.emplace_back (r * sin (theta) * cos (phi), r * sin (phi), r * cos (theta) * cos (phi));
485 normals.emplace_back (sin (theta) * cos (phi), sin (phi), cos (theta) * cos (phi));
486}

488static void model_sphere_vertices (const float r,
489 const unsigned int slices,
490 const unsigned int stacks,
491 vector<vec3> &vertices,
492 vector<vec3> &normals,
493 vector<vec2> &texture)
494{
495 // https://www.math3d.org/EumEEZBKe
496 // https://www.math3d.org/zE4n6xayX
497
498 const float s = 2.0f * (float) M_PI / (float) slices;
499 const float t = M_PI / (float) stacks;
500 const float theta = -M_PI;
501 const float phi = -M_PI / 2.0f;
502
503 auto fslices = (float) slices;
504 auto fstacks = (float) stacks;
505
506 for (unsigned int slice = 1; slice <= slices; ++slice)
507 {
508 for (unsigned int stack = 1; stack <= stacks; ++stack)
509 {
510 auto fslice = (float) slice;
511 auto fstack = (float) stack;
512
513 texture.emplace_back ((fslice - 1) / fslices, fstack / fstacks); // P1'
514 texture.emplace_back (fslice / fslices, (fstack - 1) / fstacks); // P2
515 texture.emplace_back (fslice / fslices, fstack / fstacks); // P2'
516
517 texture.emplace_back ((fslice - 1) / fslices, (fstack - 1) / fstacks); // P1
518 texture.emplace_back (fslice / fslices, (fstack - 1) / fstacks); // P2
519 texture.emplace_back ((fslice - 1) / fslices, fstack / fstacks); // P1'
520
521 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * fstack, vertices, normals); // P1'
522 model_sphere_vertex (r, theta + s * fslice, phi + t * (fstack - 1), vertices, normals); // P2
523 model_sphere_vertex (r, theta + s * fslice, phi + t * fstack, vertices, normals); // P2'
524
525 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * (fstack - 1), vertices, normals); // P1
526 model_sphere_vertex (r, theta + s * fslice, phi + t * (fstack - 1), vertices, normals); // P2
527 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * fstack, vertices, normals); // P1'
528 }
529 }
530}

552vector<array<vec3, 16>> read_Bezier (const char *const patch)
553{
554 string buffer;
555 ifstream myFile;
556
557 myFile.open (patch, ios::in | ios::out);
558 getline (myFile, buffer);
559 // Número de patches presentes no ficheiro.
560 const int n_patches = stoi (buffer);
561
562 // Vetor de vetores de índices.
563 vector<vector<int>> patches;
564
565 // Ciclo externo lê uma linha (patch) de cada vez
566 for (int j = 0; j < n_patches; j++)
567 {
568 vector<int> patchIndexes;
569 /*
570 Ciclo interno lê os índices dos pontos de controlo de cada patch, sabendo que cada
571 patch terá 16 pontos de controlo.
572 */
573 for (int i = 0; i < 15; i++)
574 {
575 getline (myFile, buffer, ',');
576 patchIndexes.push_back (stoi (buffer));
577 }
578 getline (myFile, buffer);
579 patchIndexes.push_back (stoi (buffer));
580 patches.push_back (patchIndexes);
581 }
582
583 getline (myFile, buffer);
584 // Número de pontos presentes no ficheiro.
585 const int pts = stoi (buffer);
586
587 // Vetor que guardará as coordenadas de pontos de controlo para superfície de Bézier.
588 vector<vec3> control;
589 for (int j = 0; j < pts; j++)
590 {
591 vec3 v;
592 getline (myFile, buffer, ',');
593 v[0] = stof (buffer);
594 getline (myFile, buffer, ',');
595 v[1] = stof (buffer);
596 getline (myFile, buffer);
597 v[2] = stof (buffer);
598 control.push_back (v);
599 }
600
601 /*
602 Percorrem-se os vetores que, para cada patch, guardam os seus índices de pontos de controlo.
603 Para cada patch, constroi-se um vetor com as coordenadas dos seus pontos de controlo.
604 */
605 vector<array<vec3, 16>> pointsInPatches;
606 for (auto &patche : patches)
607 {
608 array<vec3, 16> pointsInPatch{};
609 for (int j = 0; j < 16; j++)
610 {
611 pointsInPatch[j] = control[patche[j]];
612 }
613 pointsInPatches.push_back (pointsInPatch);
614 }
615
616 myFile.close ();
617
618 /*std::cout << "read_Bezier read:" << std::endl;
619 for (auto arr : pointsInPatches)
620 for (auto p : arr)
621 std::cout << glm::to_string (p) << std::endl;*/
622 return pointsInPatches;
623}

649void get_bezier_point_at (
650 const auto u,
651 const auto v,
652 const array<vec3, 16> &cp,
653 vec3 &coordinate_in_3d_space,
654 vec3 &normal)
655{
656
657 // P_u notation at (slide 17)[Curves and Surfaces]
658
659 // P(u,v) = UM(Pi0(P0u) + Pi1(P1u) + Pi2(P2u) Pi3(P3u)) based on (page 6)[CURVES AND SURFACES]
660
661 // 4 sets of control points for 4 Bézier curves
662 array<vec3, 4> C_i0 = {cp[0], cp[1], cp[2], cp[3]};
663 array<vec3, 4> C_i1 = {cp[4], cp[5], cp[6], cp[7]};
664 array<vec3, 4> C_i2 = {cp[8], cp[9], cp[10], cp[11]};
665 array<vec3, 4> C_i3 = {cp[12], cp[13], cp[14], cp[15]};
666
667 // calculate P_i(u) at each Bézier curve defined by Ci
668 vec3 P0u, P1u, P2u, P3u, _;
669 get_curve_point_at (u, Mb, C_i0, P0u, _);
670 get_curve_point_at (u, Mb, C_i1, P1u, _);
671 get_curve_point_at (u, Mb, C_i2, P2u, _);
672 get_curve_point_at (u, Mb, C_i3, P3u, _);
673
674 // Calculate vector tangent to u⃗
675 const auto VM = monic_3rd_polynomial_at (v) * Mb;
676 const auto U_prime = deriv_of_monic_3rd_polynomial_at (u);
677 const auto tangent_u =
678 (mat (C_i0) * VM[0] + mat (C_i1) * VM[1] + mat (C_i2) * VM[2] + mat (C_i3) * VM[3]) * Mb * U_prime;
679
680 vec3 Puv, tangent_v;
681 get_curve_point_at (v, Mb, {P0u, P1u, P2u, P3u}, Puv, tangent_v);
682
683 normal = normalize (cross (tangent_u, tangent_v));
684 coordinate_in_3d_space = Puv;
685}

699void get_bezier_patch (
700 const array<vec3, 16> control_points,
701 const int int_tesselation,
702 vector<vec3> &vertices,
703 vector<vec3> &normals,
704 vector<vec2> &texture)
705{
706 auto float_tesselation = (float) int_tesselation;
707 const auto step = 1.0 / float_tesselation;
708 for (int v = 0; v < int_tesselation; ++v)
709 {
710 for (int u = 0; u < int_tesselation; ++u)
711 {
712 for (auto e : {
713 // upper triangle
714 array<float, 2>{0, 1},
715 array<float, 2>{0, 0},
716 array<float, 2>{1, 0},
717 // lower triangle
718 array<float, 2>{1, 0},
719 array<float, 2>{1, 1},
720 array<float, 2>{0, 1},
721 })
722 {
723 vec3 vertex, normal;
724 get_bezier_point_at (u * step + step * e[0], v * step + step * e[1], control_points, vertex, normal);
725
726 vertices.push_back (vertex);
727 normals.push_back (normal);
728 texture.emplace_back (-(u * step + step * e[0]), -(v * step + step * e[1]));
729 }
730 }
731 }

733 for (auto p : pontos)
734 std::cout << glm::to_string (p) << std::endl;*/
735}

108void model_plane_vertices (const float length,
109 const unsigned int divisions,
110 vector<vec3> &vertices,
111 vector<vec3> &normals,
112 vector<vec2> &texture)
113{
114 const float o = -length / 2.0f;
115 const float d = length / (float) divisions;
116
117 for (unsigned int uidiv1 = 1; uidiv1 <= divisions; ++uidiv1)
118 {
119 for (unsigned int uidiv2 = 1; uidiv2 <= divisions; ++uidiv2)
120 {
121 auto const fdiv1 = (float) uidiv1;
122 auto const fdiv2 = (float) uidiv2;
123 auto const fdivisions = (float) divisions;
124
125 for (auto e : {
126 array<float, 2>{-1, -1},//P1
127 array<float, 2>{-1, 0},//P1'z
128 array<float, 2>{0, -1},//P1'x
129
130 array<float, 2>{0, -1},//P1'x
131 array<float, 2>{-1, 0},//P1'z
132 array<float, 2>{0, 0}})//P2
133 {
134 vertices.emplace_back (o + d * (fdiv1 + e[0]), 0, o + d * (fdiv2 + e[1]));
135 normals.emplace_back (0, 1, 0);
136 texture.emplace_back ((fdiv1 + e[0]) / fdivisions, (fdiv2 + e[1]) / fdivisions);
137 }
138
139
140 /*Cull face*/
141 for (auto e : {
142 array<float, 2>{-1.0f, 0.0f},
143 array<float, 2>{-1.0f, -1.0f},
144 array<float, 2>{0.0f, -1.0f},
145
146 array<float, 2>{-1.0f, 0.0f},
147 array<float, 2>{0.0f, -1.0f},
148 array<float, 2>{0.0f, 0.0f}})
149 {
150 vertices.emplace_back (o + d * (fdiv1 + e[0]), 0, o + d * (fdiv2 + e[1]));
151 normals.emplace_back (0, -1, 0);
152 texture.emplace_back ((fdiv1 + e[0]) / fdivisions, (fdiv2 + e[1]) / fdivisions);

369template<typename T>
370 requires arithmetic<T>
371static inline void
372model_cone_vertex (const T r,
373 const T height,
374 const T theta,
375 const T h,
376 vector<vec3> &vertices)
377{
378 /*
379 x = r (h/height) cos()
380 y = 2 (height + h)
381 z = r (h/height) sin()
382
383 r 0
384 {- + is : s = 2/slices i {0,...,slices} }
385 h {-height + jt : t = height/stacks j {0,...,stacks} }
386
387 check:
388 1. https://www.math3d.org/7oeSkmuns
389 */
390 vertices.emplace_back (r * h / height * cos (theta), height + h, r * h / height * sin (theta));
391}

393template<typename T>
394 requires arithmetic<T>
395void model_cone_vertices (const T radius,
396 const T height,
397 const unsigned int slices,
398 const unsigned int stacks,
399 vector<vec3> &vertices,
400 vector<vec3> &normals,
401 vector<vec2> &texture)
402{
403
404 const T s = 2 * M_PI / (float) slices;
405 const T t = height / (float) stacks;
406
407 const T theta_0 = -M_PI;
408 const T h_0 = -height;
409
410 auto const fslices = (float) slices;
411 auto const fstacks = (float) stacks;
412
413 for (unsigned int slice = 1; slice <= slices; ++slice)
414 {
415 for (unsigned int stack = 1; stack <= stacks; ++stack)
416 {
417 auto const fslice = (float) slice;
418 auto const fstack = (float) stack;
419
420 //base
421 vertices.emplace_back (0, 0, 0); //O
422 texture.emplace_back (0, 0);
423 normals.emplace_back (0, -1, 0);
424 for (auto e : {
425 -1.0f,//P1
426 .0f //P2
427 })
428 {
429 model_cone_vertex (radius, height, theta_0 + s * (fslice + e), h_0, vertices); //P1
430 normals.emplace_back (0, -1, 0);
431 }
432
433 texture.emplace_back (-1, 0);
434 texture.emplace_back (0, 0);
435
436 int q = 0;
437 for (auto e : {
438 array<float, 2>{0, -1},
439 array<float, 2>{-1, 0},
440 array<float, 2>{0, 0},
441
442 array<float, 2>{0, -1},
443 array<float, 2>{-1, -1},
444 array<float, 2>{-1, 0},
445 })
446 {
447 model_cone_vertex (radius, height,
448 theta_0 + s * (fslice + e[0]),
449 h_0 + t * (fstack + e[1]), vertices);
450 if (q % 3 == 2)
451 {
452 const auto P1 = vertices.end ()[-2];
453 const auto P2 = vertices.end ()[-3];
454 const auto P1_prime = vertices.end ()[-1];
455 for (auto _ = 0; _ < 3; ++_)
456 normals.emplace_back (normalize (cross (P2 - P1_prime, P1 - P1_prime)));
457 }
458
459 texture.emplace_back (fslice / fslices, fstack / fstacks);
460 ++q;
461 }
462 }
463 }
464}

498static inline void
499model_sphere_vertex (const float r,
500 const float theta,
501 const float phi,
502 vector<vec3> &vertices,
503 vector<vec3> &normals)
504{
505 /*
506 x = r sin()cos()
507 y = r sin()
508 z = r cos()cos()
509
510 r 0
511 {- + is : s = 2/slices i {0,...,slices} }
512 {-/2 + jt : t = /stacks j {0,...,stacks} }
513
514 check
515 1. https://www.math3d.org/EumEEZBKe
516 2. https://www.math3d.org/zE4n6xayX
517 */
518 vertices.emplace_back (r * sin (theta) * cos (phi), r * sin (phi), r * cos (theta) * cos (phi));
519 normals.emplace_back (sin (theta) * cos (phi), sin (phi), cos (theta) * cos (phi));
520}

522static void model_sphere_vertices (const float r,
523 const unsigned int slices,
524 const unsigned int stacks,
525 vector<vec3> &vertices,
526 vector<vec3> &normals,
527 vector<vec2> &texture)
528{
529 // https://www.math3d.org/EumEEZBKe
530 // https://www.math3d.org/zE4n6xayX
531
532 const float s = 2.0f * (float) M_PI / (float) slices;
533 const float t = M_PI / (float) stacks;
534 const float theta = -M_PI;
535 const float phi = -M_PI / 2.0f;
536
537 auto fslices = (float) slices;
538 auto fstacks = (float) stacks;
539
540 for (unsigned int slice = 1; slice <= slices; ++slice)
541 {
542 for (unsigned int stack = 1; stack <= stacks; ++stack)
543 {
544 auto fslice = (float) slice;
545 auto fstack = (float) stack;
546
547 texture.emplace_back ((fslice - 1) / fslices, fstack / fstacks); // P1'
548 texture.emplace_back (fslice / fslices, (fstack - 1) / fstacks); // P2
549 texture.emplace_back (fslice / fslices, fstack / fstacks); // P2'
550
551 texture.emplace_back ((fslice - 1) / fslices, (fstack - 1) / fstacks); // P1
552 texture.emplace_back (fslice / fslices, (fstack - 1) / fstacks); // P2
553 texture.emplace_back ((fslice - 1) / fslices, fstack / fstacks); // P1'
554
555 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * fstack, vertices, normals); // P1'
556 model_sphere_vertex (r, theta + s * fslice, phi + t * (fstack - 1), vertices, normals); // P2
557 model_sphere_vertex (r, theta + s * fslice, phi + t * fstack, vertices, normals); // P2'
558
559 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * (fstack - 1), vertices, normals); // P1
560 model_sphere_vertex (r, theta + s * fslice, phi + t * (fstack - 1), vertices, normals); // P2
561 model_sphere_vertex (r, theta + s * (fslice - 1), phi + t * fstack, vertices, normals); // P1'
562 }
563 }
564}

588vector<array<vec3, 16>> read_Bezier (const char *const patch)
589{
590 string buffer;
591 ifstream myFile;
592
593 myFile.open (patch, ios::in | ios::out);
594 getline (myFile, buffer);
595 // Nmero de patches presentes no ficheiro.
596 const int n_patches = stoi (buffer);
597
598 // Vetor de vetores de ndices.
599 vector<vector<int>> patches;
600
601 // Ciclo externo l uma linha (patch) de cada vez
602 for (int j = 0; j < n_patches; j++)
603 {
604 vector<int> patchIndexes;
605 /*
606 Ciclo interno l os ndices dos pontos de controlo de cada patch, sabendo que cada
607 patch ter 16 pontos de controlo.
608 */
609 for (int i = 0; i < 15; i++)
610 {
611 getline (myFile, buffer, ',');
612 patchIndexes.push_back (stoi (buffer));
613 }
614 getline (myFile, buffer);
615 patchIndexes.push_back (stoi (buffer));
616 patches.push_back (patchIndexes);
617 }
618
619 getline (myFile, buffer);
620 // Nmero de pontos presentes no ficheiro.
621 const int pts = stoi (buffer);
622
623 // Vetor que guardar as coordenadas de pontos de controlo para superfcie de Bzier.
624 vector<vec3> control;
625 for (int j = 0; j < pts; j++)
626 {
627 vec3 v;
628 getline (myFile, buffer, ',');
629 v[0] = stof (buffer);
630 getline (myFile, buffer, ',');
631 v[1] = stof (buffer);
632 getline (myFile, buffer);
633 v[2] = stof (buffer);
634 control.push_back (v);
635 }
636
637 /*
638 Percorrem-se os vetores que, para cada patch, guardam os seus ndices de pontos de controlo.
639 Para cada patch, constroi-se um vetor com as coordenadas dos seus pontos de controlo.
640 */
641 vector<array<vec3, 16>> pointsInPatches;
642 for (auto &patche : patches)
643 {
644 array<vec3, 16> pointsInPatch{};
645 for (int j = 0; j < 16; j++)
646 {
647 pointsInPatch[j] = control[patche[j]];
648 }
649 pointsInPatches.push_back (pointsInPatch);
650 }
651
652 myFile.close ();
653
654 /*std::cout << "read_Bezier read:" << std::endl;
655 for (auto arr : pointsInPatches)
656 for (auto p : arr)
657 std::cout << glm::to_string (p) << std::endl;*/
658 return pointsInPatches;
659}

689 vec3 &coordinate_in_3d_space,
690 vec3 &normal)
691{
692
693 // P_u notation at (slide 17)[Curves and Surfaces]
694
695 // P(u,v) = UM(Pi0(P0u) + Pi1(P1u) + Pi2(P2u) Pi3(P3u)) based on (page 6)[CURVES AND SURFACES]
696
697 // 4 sets of control points for 4 Bzier curves
698 array<vec3, 4> C_i0 = {cp[0], cp[1], cp[2], cp[3]};
699 array<vec3, 4> C_i1 = {cp[4], cp[5], cp[6], cp[7]};
700 array<vec3, 4> C_i2 = {cp[8], cp[9], cp[10], cp[11]};
701 array<vec3, 4> C_i3 = {cp[12], cp[13], cp[14], cp[15]};
702
703 // calculate P_i(u) at each Bzier curve defined by Ci
704 vec3 P0u, P1u, P2u, P3u, _;
705 get_curve_point_at (u, Mb, C_i0, P0u, _);
706 get_curve_point_at (u, Mb, C_i1, P1u, _);
707 get_curve_point_at (u, Mb, C_i2, P2u, _);
708 get_curve_point_at (u, Mb, C_i3, P3u, _);
709
710 // Calculate vector tangent to u
711 const auto VM = monic_3rd_polynomial_at (v) * Mb;
712 const auto U_prime = deriv_of_monic_3rd_polynomial_at (u);
713 const auto tangent_u =
714 (mat (C_i0) * VM[0] + mat (C_i1) * VM[1] + mat (C_i2) * VM[2] + mat (C_i3) * VM[3]) * Mb * U_prime;
715
716 vec3 Puv, tangent_v;
717 get_curve_point_at (v, Mb, {P0u, P1u, P2u, P3u}, Puv, tangent_v);
718
719 normal = normalize (cross (tangent_u, tangent_v));
720 coordinate_in_3d_space = Puv;
721}

  46 void model_plane_vertices(const float length, const unsigned int divisions, float *points) {
  47     const float o = -length / 2.0f;
  48     const float d = length / (float) divisions;
  49 
  50     unsigned int pos = 0;
  51 
  52     for (unsigned int m = 1; m <= divisions; m++) {
  53         for (unsigned int n = 1; n <= divisions; n++) {
  54             float i = (float) m;
  55             float j = (float) n;
  56 
  57             points_vertex(o + d * (i - 1), 0, o + d * (j - 1), &pos, points); //P1
  58             points_vertex(o + d * (i - 1), 0, o + d * j, &pos, points); //P1'z
  59             points_vertex(o + d * i, 0, o + d * (j - 1), &pos, points); //P1'x
  60 
  61             points_vertex(o + d * i, 0, o + d * (j - 1), &pos, points); //P1'x
  62             points_vertex(o + d * (i - 1), 0, o + d * j, &pos, points); //P1'z
  63             points_vertex(o + d * i, 0, o + d * j, &pos, points); //P2
  64 
  65             /*Cull face*/
  66             points_vertex(o + d * (i - 1), 0, o + d * j, &pos, points); //P1'z
  67             points_vertex(o + d * (i - 1), 0, o + d * (j - 1), &pos, points); //P1
  68             points_vertex(o + d * i, 0, o + d * (j - 1), &pos, points); //P1'x
  69 
  70             points_vertex(o + d * (i - 1), 0, o + d * j, &pos, points); //P1'z
  71             points_vertex(o + d * i, 0, o + d * (j - 1), &pos, points); //P1'x
  72             points_vertex(o + d * i, 0, o + d * j, &pos, points); //P2
  73         }
  74     }

 171 /*!
 172  * \f{aligned}{
 173  * x &= r\frac{h}{\textrm{height}}  \cos()\\[2em]
 174  * y &= h + \textrm{height}\\[2em]
 175  * z &= r\frac{h}{\textrm{height}}  \sin()
 176  * \f}\n
 177  *
 178  * \f{aligned}{
 179  *  r & 0\\
 180  *   & \left\{-      + is : s = \frac{2}{\textrm{slices}}       i  \{0,...,\textrm{slices}\} \right\}\\
 181  *  h & \left\{- \textrm{height} + jt : t =  \frac{\textrm{height}}{\textrm{stacks}}  j  \{0,...,\textrm{stacks}\} \right\}
 182  *  \f}
 183  *
 184  *  See the [3d model](https://www.math3d.org/7oeSkmuns).
 185  */
 186 static inline void
 187 model_cone_vertex(const float r, const float height, const float theta, const float h, unsigned int *pos,
 188                   float *points) {
 189     /*
 190        x = r  (h/height)  cos()
 191        y = 2  (height + h)
 192        z = r  (h/height)  sin()
 193 
 194        r  0
 195          {-      + is : s = 2/slices       i  {0,...,slices} }
 196        h  {-height + jt : t =  height/stacks  j  {0,...,stacks} }
 197 
 198        check:
 199            1. https://www.math3d.org/7oeSkmuns
 200      */
 201     points_vertex(r * h / height * cos(theta), height + h, r * h / height * sin(theta), pos, points);
 202 }

 204 void model_cone_vertices(const float r, const float height, const unsigned int slices, const unsigned int stacks,
 205                          float points[]) {
 206 
 207     const float s = 2.0f * (float) M_PI / (float) slices;
 208     const float t = height / (float) stacks;
 209     const float theta = -M_PI;
 210     const float h = -height;
 211 
 212     unsigned int pos = 0;
 213 
 214     for (unsigned int m = 1; m <= slices; m++) {
 215         for (unsigned int n = 1; n <= stacks; n++) {
 216             float i = (float) m;
 217             float j = (float) n;
 218 
 219             //base
 220             points_vertex(0, 0, 0, &pos, points); //O
 221             model_cone_vertex(r, height, theta + s * (i - 1), h, &pos, points); //P1
 222             model_cone_vertex(r, height, theta + s * i, h, &pos, points); //P2
 223 
 224             model_cone_vertex(r, height, theta + s * i, h + t * (j - 1), &pos, points); // P2
 225             model_cone_vertex(r, height, theta + s * (i - 1), h + t * j, &pos, points); // P1'
 226             model_cone_vertex(r, height, theta + s * i, h + t * j, &pos, points); //P2'
 227 
 228             model_cone_vertex(r, height, theta + s * i, h + t * (j - 1), &pos, points); // P2
 229             model_cone_vertex(r, height, theta + s * (i - 1), h + t * (j - 1), &pos, points); // P1
 230             model_cone_vertex(r, height, theta + s * (i - 1), h + t * j, &pos, points); // P1'
 231         }
 232     }
 233 }

 252 static inline void
 253 model_sphere_vertex(const float r, const float theta, const float phi, unsigned int *pos, float *points) {
 254     /*
 255         x = r  sin()cos()
 256         y = r  sin()
 257         z = r  cos()cos()
 258 
 259         r  0
 260           {- +   is : s = 2/slices  i  {0,...,slices} }
 261           {-/2 + jt : t =  /stacks  j  {0,...,stacks} }
 262 
 263         check
 264             1. https://www.math3d.org/EumEEZBKe
 265             2. https://www.math3d.org/zE4n6xayX
 266      */
 267     points_vertex(r * sin(theta) * cos(phi), r * sin(phi), r * cos(theta) * cos(phi), pos, points);
 268 }
 269 
 270 static void model_sphere_vertices(const float r, const unsigned int slices, const unsigned int stacks, float points[]) {
 271     // https://www.math3d.org/EumEEZBKe
 272     // https://www.math3d.org/zE4n6xayX
 273 
 274     const float s = 2.0f * (float) M_PI / (float) slices;
 275     const float t = M_PI / (float) stacks;
 276     const float theta = -M_PI;
 277     const float phi = -M_PI / 2.0f;
 278 
 279     unsigned int pos = 0;
 280 
 281     for (unsigned int m = 1; m <= slices; m++) {
 282         for (unsigned int n = 1; n <= stacks; n++) {
 283             float i = (float) m;
 284             float j = (float) n;
 285 
 286             model_sphere_vertex(r, theta + s * (i - 1), phi + t * j, &pos, points); // P1'
 287             model_sphere_vertex(r, theta + s * i, phi + t * (j - 1), &pos, points); // P2
 288             model_sphere_vertex(r, theta + s * i, phi + t * j, &pos, points); //P2'
 289 
 290             model_sphere_vertex(r, theta + s * (i - 1), phi + t * (j - 1), &pos, points); // P1
 291             model_sphere_vertex(r, theta + s * i, phi + t * (j - 1), &pos, points); // P2
 292             model_sphere_vertex(r, theta + s * (i - 1), phi + t * j, &pos, points); // P1'
 293         }
 294     }
 295 }

 313 int main(int argc, char *argv[]) {
 314     if (argc < 5) { printf("Not enough arguments"); }
 315     else {
 316         const char *filepath = argv[argc - 1];
 317         const char *polygon = argv[1];
 318 
 319         if (!strcmp(PLANE, polygon))
 320             model_plane_write(filepath, strtof(argv[2], nullptr), strtoul(argv[3], nullptr, 10));
 321         if (!strcmp(CUBE, polygon)) model_cube_write(filepath, atof(argv[2]), strtoul(argv[3], nullptr, 10));
 322         if (!strcmp(CONE, polygon))
 323             model_cone_write(filepath, atof(argv[2]), atof(argv[3]), atoi(argv[4]), atoi(argv[5]));
 324         if (!strcmp(SPHERE, polygon)) model_sphere_write(filepath, atof(argv[2]), atoi(argv[3]), atoi(argv[4]));
 325     }
 326     return 1;

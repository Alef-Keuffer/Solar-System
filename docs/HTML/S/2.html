  57 void operations_push_transforms(tinyxml2::XMLElement *transforms, std::vector<float> *operations) {
  58     tinyxml2::XMLElement *transform = transforms->FirstChildElement();
  59 
  60     do {
  61         const char *transformValue = transform->Value();
  62 
  63         if (!strcmp("translate", transformValue))
  64             operations->push_back(TRANSLATE);
  65         else if (!strcmp("rotate", transformValue))
  66             operations->push_back(ROTATE);
  67         else if (!strcmp("scale", transformValue))
  68             operations->push_back(SCALE);
  69         else {
  70             fprintf(stderr, "Unknown transform: \"%s\"", transformValue);
  71             exit(1);
  72         }
  73 
  74         operations_push_transform_attributes(transform, operations);
  75 
  76     } while ((transform = transform->NextSiblingElement()));
  77 }

 110 /*! @addtogroup Groups
 111  * @{*/
 112 void operations_push_groups(tinyxml2::XMLElement *group, std::vector<float> *operations) {
 113     operations->push_back(BEGIN_GROUP);
 114 
 115     tinyxml2::XMLElement *transforms = group->FirstChildElement("transform");
 116     tinyxml2::XMLElement *models = group->FirstChildElement("models");
 117 
 118     if (transforms) operations_push_transforms(transforms, operations);
 119     if (models) operations_push_models(models, operations);
 120 
 121     tinyxml2::XMLElement *childGroup = group->FirstChildElement("group");
 122     if (childGroup)
 123         do operations_push_groups(childGroup, operations);
 124         while ((childGroup = childGroup->NextSiblingElement("group")));
 125 
 126     operations->push_back(END_GROUP);
 127 }

 133 void operations_load_xml(const char *filename, std::vector<float> *operations) {
 134     tinyxml2::XMLDocument doc;
 135 
 136     if (doc.LoadFile(filename)) {
 137         fprintf(stderr, "%s", doc.ErrorName());
 138         exit(1);
 139     }
 140 
 141     tinyxml2::XMLElement *world = doc.FirstChildElement("world");
 142 
 143     /*camera*/
 144     tinyxml2::XMLElement *camera = world->FirstChildElement("camera");
 145 
 146     tinyxml2::XMLElement *position = camera->FirstChildElement("position");
 147     operations_push_transform_attributes(position, operations);
 148 
 149     tinyxml2::XMLElement *lookAt = camera->FirstChildElement("lookAt");
 150     operations_push_transform_attributes(lookAt, operations);
 151 
 152     tinyxml2::XMLElement *up = camera->FirstChildElement("up");
 153     if (up) operations_push_transform_attributes(up, operations);
 154     else operations->insert(operations->end(), {0, 1, 0});
 155 
 156     tinyxml2::XMLElement *projection = camera->FirstChildElement("projection");
 157     if (projection) {
 158         operations->push_back(projection->FloatAttribute("fov"));
 159         operations->push_back(projection->FloatAttribute("near"));
 160         operations->push_back(projection->FloatAttribute("far"));
 161     } else operations->insert(operations->end(), {60, 1, 1000});
 162     /*end of camera*/
 163 
 164     // groups
 165     tinyxml2::XMLElement *group = world->FirstChildElement("group");
 166     operations_push_groups(group, operations);
 167 }

 172 void operations_print(std::vector<float> *operations) {
 173     unsigned int i = 0;
 174     fprintf(stderr,
 175             "POSITION(%.2f %.2f %.2f)\n",
 176             operations->at(i),
 177             operations->at(i + 1),
 178             operations->at(i + 2));
 179     i += 3;
 180     fprintf(stderr,
 181             "LOOK_AT(%.2f %.2f %.2f)\n",
 182             operations->at(i),
 183             operations->at(i + 1),
 184             operations->at(i + 2));
 185     i += 3;
 186     fprintf(stderr,
 187             "UP(%.2f %.2f %.2f)\n",
 188             operations->at(i),
 189             operations->at(i + 1),
 190             operations->at(i + 2));
 191     i += 3;
 192     fprintf(stderr,
 193             "PROJECTION(%.2f %.2f %.2f)\n",
 194             operations->at(i),
 195             operations->at(i + 1),
 196             operations->at(i + 2));
 197 
 198     for (i += 3; i < operations->size(); i++) {
 199         switch ((int) operations->at(i)) {
 200             case ROTATE:
 201                 fprintf(stderr,
 202                         "ROTATE(%.2f %.2f %.2f %.2f)\n",
 203                         operations->at(i + 1),
 204                         operations->at(i + 2),
 205                         operations->at(i + 3),
 206                         operations->at(i + 4));
 207                 i += 4;
 208                 continue;
 209             case TRANSLATE:
 210                 fprintf(stderr,
 211                         "TRANSLATE(%.2f %.2f %.2f)\n",
 212                         operations->at(i + 1),
 213                         operations->at(i + 2),
 214                         operations->at(i + 3));
 215                 i += 3;
 216                 continue;
 217             case SCALE:
 218                 fprintf(stderr,
 219                         "SCALE(%.2f %.2f %.2f)\n",
 220                         operations->at(i + 1),
 221                         operations->at(i + 2),
 222                         operations->at(i + 3));
 223                 i += 3;
 224                 continue;
 225             case BEGIN_GROUP:
 226                 fprintf(stderr, "BEGIN_GROUP\n");
 227                 continue;
 228             case END_GROUP:
 229                 fprintf(stderr, "END_GROUP\n");
 230                 continue;
 231             case LOAD_MODEL:
 232                 int stringSize = (int) operations->at(i + 1);
 233                 char model[stringSize + 1];
 234 
 235                 int j;

 237                     model[j] = (char) operations->at(i + 2 + j);
 238

 242                 i += 1 + j - 1; //just to be explicit
 243                 continue;
 244         }
 245     }

100template<typename T>
101T getAttribute (const XMLElement &e, const string &attributeName)
102{
103 T _attributeValue;
104 XMLError error;
105
106 if constexpr(std::is_same<string, T>::value)
107 {
108 const char **attributeValue;
109 error = e.QueryStringAttribute (attributeName.c_str (), attributeValue);
110 _attributeValue = string{*attributeValue};
111 }
112 else if constexpr(std::is_same<float, T>::value)
113 {
114 float attributeValue;
115 error = e.QueryFloatAttribute (attributeName.c_str (), &attributeValue);
116 _attributeValue = attributeValue;
117 }
118 else if constexpr(std::is_same<bool, T>::value)
119 {
120 bool attributeValue;
121 error = e.QueryBoolAttribute (attributeName.c_str (), &attributeValue);
122 _attributeValue = attributeValue;
123 }
124
125 crashIfFailedAttributeQuery (error, e, attributeName);
126 return _attributeValue;
127}

190void operations_push_any (const XMLElement *element, vector<float> &operations)
191{
192 static string generatorExecutable;
193 static bool isUsingGenerator = false;
194 do
195 {
196 auto elementEnum = elementToEnum (element->Value ());
197 switch (elementEnum)
198 {
199 case GROUP:
200 {
201 switch (element->ClosingType ())
202 {
203 case XMLElement::OPEN:
204 operations.push_back (BEGIN_GROUP);
205 break;
206 case XMLElement::CLOSED:
207 break;
208 case XMLElement::CLOSING:
209 operations.push_back (END_GROUP); //never happens...i think
210 break;
211 }
212 break;
213 }
214 case GENERATOR:
215 {
216 if (element->Attribute ("dir"))
217 {
218 generatorExecutable = getAttribute<string> (*element, "dir");
219 crashIfFileDoesNotExist (generatorExecutable);
220 cerr << "[parsing] using generator " << generatorExecutable << endl;
221 isUsingGenerator = true;
222 }
223 else if (isUsingGenerator)
224 {
225 auto generator_argv = getAttribute<string> (*element, "argv");
226 std::stringstream command;
227 command << std::filesystem::current_path ()
228 << "/" << generatorExecutable << " " << generator_argv;
229 if (system (command.str ().data ()))
230 {
231 cerr << "[parsing] generator failed at line " << element->GetLineNum () << endl;
232 exit (EXIT_FAILURE);
233 }
234 }
235 break;
236 }
237 case MODEL:
238 {
239 switch (element->ClosingType ())
240 {
241 case XMLElement::OPEN:
242 operations.push_back (BEGIN_MODEL);
243 case XMLElement::CLOSED:
244 operations_push_string_attribute (*element, operations, "file");
245 break;
246 case XMLElement::CLOSING: // never happens...i think
247 operations.push_back (END_MODEL);
248 break;
249 }
250 break;
251 }
252 case ROTATE:
253 {
254 if (element->Attribute ("time"))
255 operations.push_back (EXTENDED_ROTATE);
256 else
257 operations.push_back (ROTATE);
258 pushFloatAttributes (*element, operations);
259 break;
260 }
261 case TRANSLATE:
262 {
263 switch (element->ClosingType ())
264 {
265 // ⟨extended_translation⟩ ::= ⟨EXTENDED_TRANSLATE⟩⟨time⟩⟨align⟩⟨number_of_points⟩⟨vec3f⟩⁺
266 case XMLElement::OPEN:
267 {
268 operations.push_back (EXTENDED_TRANSLATE);
269
270 auto time = getAttribute<float> (*element, "time");
271 auto align = getAttribute<bool> (*element, "align");
272
273 operations.insert (operations.end (), {time, (float) align});
274 operations.push_back (0); // create space to insert the number of points
275 auto index_of_number_of_points = operations.size () - 1;
276
277 auto number_of_points = 0;
278 for (auto child = element->FirstChildElement ("point"); child;
279 child = child->NextSiblingElement ("point"))
280 {
281 ++number_of_points;
282 pushFloatAttributes (*child, operations);
283 }
284 operations[index_of_number_of_points] = (float) number_of_points;
285 }
286 break;
287 // ⟨simple_translation⟩ ::= ⟨TRANSLATE⟩⟨vec3f⟩
288 case XMLElement::CLOSED:
289 {
290 operations.push_back (TRANSLATE);
291 pushFloatAttributes (*element, operations);
292 }
293 break;
294 case XMLElement::CLOSING:
295 break;
296 }
297 break;
298 }
299 case TEXTURE: // ⟨texture⟩ ::= ⟨TEXTURE⟩ ⟨number of characters⟩ ⟨char⟩⁺
300 {
301 operations.push_back (elementEnum);
302 operations_push_string_attribute (*element, operations, "file");
303 break;
304 }
305 // (⟨DIFFUSE⟩ | ⟨AMBIENT⟩ | ⟨SPECULAR⟩ | ⟨EMISSIVE⟩) ⟨nvec3f⟩
306 case DIFFUSE:
307 case AMBIENT:
308 case SPECULAR:
309 case EMISSIVE:
310 {
311 operations.push_back (elementEnum);
312 const float RGB_MAX = 255.0f;
313 auto R = getAttribute<float> (*element, "R");
314 auto G = getAttribute<float> (*element, "G");
315 auto B = getAttribute<float> (*element, "B");
316 assert (R >= 0 && G >= 0 && B >= 0);
317 assert (R <= RGB_MAX && G <= RGB_MAX && B <= RGB_MAX);
318 operations.insert (operations.end (), {R / RGB_MAX, G / RGB_MAX, B / RGB_MAX});
319 break;
320 }
321 case LIGHT: // ⟨point⟩ | ⟨directional⟩ | ⟨spotlight⟩
322 {
323 elementEnum = elementToEnum (getAttribute<string> (*element, "type"));
324 if (elementEnum == OTHER)
325 elementEnum = POINT;
326 }
327 case SHININESS: // SHININESS⟩ ⟨float ∈ [0, 128]⟩
328 case SCALE: // ⟨scaling⟩ ::= ⟨SCALE⟩⟨vec3f⟩
329 case POINT: // ⟨point⟩ ::= ⟨POINT⟩⟨vec3f⟩
330 case DIRECTIONAL: // ⟨directional⟩ ::= ⟨DIRECTIONAL⟩⟨vec3f⟩
331 case SPOTLIGHT: // ⟨spotlight⟩ ::= ⟨SPOTLIGHT⟩⟨vec3f⟩⟨vec3f⟩⟨cutoff⟩
332 case POSITION: // ⟨position⟩ ::= ⟨POSITION⟩ ⟨vec3f⟩
333 case LOOK_AT: // ⟨lookAt⟩ ::= ⟨LOOK_AT⟩ ⟨vec3f⟩
334 case UP: // ⟨up⟩ ::= ⟨UP⟩ ⟨vec3f⟩
335 case PROJECTION: // ⟨projection⟩ ::= ⟨PROJECTION⟩ ⟨vec3f⟩
336 {
337 operations.push_back (elementEnum);
338 pushFloatAttributes (*element, operations);
339 }
340 break;
341 }
342
343 auto child = element->FirstChildElement ();
344 if (child != nullptr)
345 operations_push_any (child, operations);
346
347 switch (elementEnum)
348 {
349 case MODEL:
350 if (element->ClosingType () == tinyxml2::XMLElement::OPEN)
351 operations.push_back (END_MODEL);
352 break;
353 case GROUP:
354 operations.push_back (END_GROUP);
355 break;
356 }
357 }
358 while ((element = element->NextSiblingElement ()) != nullptr);
359}

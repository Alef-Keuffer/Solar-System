201void fpsCamera ()
202{
203 const auto TO_RADIANS = M_PI / 180.0;
204 const float sens = globalFpsSens;
205
206 if (globalMotion.Forward)
207 {
208 globalFpsCam.x += cos ((globalFpsYaw + 90) * TO_RADIANS) / sens;
209 globalFpsCam.z -= sin ((globalFpsYaw + 90) * TO_RADIANS) / sens;
210 }
211 if (globalMotion.Backward)
212 {
213 globalFpsCam.x += cos ((globalFpsYaw + 90 + 180) * TO_RADIANS) / sens;
214 globalFpsCam.z -= sin ((globalFpsYaw + 90 + 180) * TO_RADIANS) / sens;
215 }
216 if (globalMotion.Left)
217 {
218 globalFpsCam.x += cos ((globalFpsYaw + 90 + 90) * TO_RADIANS) / sens;
219 globalFpsCam.z -= sin ((globalFpsYaw + 90 + 90) * TO_RADIANS) / sens;
220 }
221 if (globalMotion.Right)
222 {
223 globalFpsCam.x += cos ((globalFpsYaw + 90 - 90) * TO_RADIANS) / sens;
224 globalFpsCam.z -= sin ((globalFpsYaw + 90 - 90) * TO_RADIANS) / sens;
225 }
226 if (globalMotion.Down)
227 {
228 globalFpsCam.y += 1 / sens;
229 }
230 if (globalMotion.Up)
231 {
232 globalFpsCam.y -= 1 / sens;
233 }
234 /*
235 * limit the values of pitch
236 * between -60 and 70
237 */
238 if (globalFpsPitch >= 70)
239 globalFpsPitch = 70;
240 if (globalFpsPitch <= -60)
241 globalFpsPitch = -60;
242
243 glRotatef (-globalFpsPitch, 1.0, 0.0, 0.0); // Along X axis
244 glRotatef (-globalFpsYaw, 0.0, 1.0, 0.0); // Along Y axis
245 glTranslatef (-globalFpsCam.x, globalFpsCam.y, -globalFpsCam.z);
246}

406void
407spherical2Cartesian (
408 const vec3 &origin,
409 const vec3 &polar,
410 vec3 &cartesian)
411{
412 // https://www.wikiwand.com/en/Spherical_coordinate_system
413 cartesian.x = polar.x * sin (polar.y) * cos (polar.z) + origin.x;
414 cartesian.y = polar.x * cos (polar.y) + origin.y;
415 cartesian.z = polar.x * sin (polar.y) * sin (polar.z) + origin.z;
416}

427void
428cartesian2Spherical (
429 const vec3 &o,
430 const vec3 &c,
431 vec3 &p)
432{
433 // https://www.wikiwand.com/en/Spherical_coordinate_system
434 p.x = sqrt (pow ((c.x - o.x), 2) + pow ((c.y - o.y), 2) + pow ((c.z - o.z), 2));
435 p.y = acos ((c.y - o.y) / (p.x));
436 p.z = atan2 ((c.z - o.z), (c.x - o.z));
437}

857struct model allocModel (const char *const model3dFilePath)
858{
859 FILE *fp = fopen (model3dFilePath, "r");
860 if (!fp)
861 {
862 cerr << "failed to open model: " << model3dFilePath << endl;
863 exit (EXIT_FAILURE);
864 }
865
866 cerr << "[allocModel] model file = " << model3dFilePath << endl;
867 // read number of vertices
868 GLsizei nVertices;
869 fread (&nVertices, sizeof (nVertices), 1, fp);
870 cerr << "[allocModel] nVertices = " << nVertices << endl;
871
872 // read vertices
873 auto *arrayOfVertices = (float *) malloc (3 * nVertices * sizeof (float));
874 const size_t nVerticesRead = fread (arrayOfVertices, 3 * sizeof (float), nVertices, fp);
875 if (nVerticesRead != nVertices)
876 {
877 cerr << nVerticesRead << " = nVerticesRead != nVertices = " << nVertices << endl;
878 exit (EXIT_FAILURE);
879 }
880
881 // read normals
882 auto *arrayOfNormals = (float *) malloc (3 * nVertices * sizeof (float));
883 const size_t nNormalsRead = fread (arrayOfNormals, 3 * sizeof (float), nVertices, fp);
884 if (nNormalsRead != nVertices)
885 {
886 cerr << nNormalsRead << " = nNormalsRead != nVertices = " << nVertices << endl;
887 exit (EXIT_FAILURE);
888 }
889
890 // read texture coordinates
891 auto *arrayOfTextureCoordinates = (float *) malloc (2 * nVertices * sizeof (float));
892 const size_t nTextureCoordinatesRead = fread (arrayOfTextureCoordinates, 2 * sizeof (float), nVertices, fp);
893 if (nTextureCoordinatesRead != nVertices)
894 {
895 fprintf (stderr, "%zu = nTextureCoordinatesRead != nVertices = %ul", nTextureCoordinatesRead, nVertices);
896 exit (EXIT_FAILURE);
897 }
898
899 fclose (fp);

901 struct model model;
902 model.nVertices = nVertices;
903
904 // vertices buffer object array
905 const GLsizei sizeOfVertexArray = (GLsizei) sizeof (arrayOfVertices[0]) * 3 * model.nVertices;
906 glGenBuffers (1, &model.vbo);
907 glBindBuffer (GL_ARRAY_BUFFER, model.vbo);
908 glBufferData (GL_ARRAY_BUFFER, sizeOfVertexArray, arrayOfVertices, GL_STATIC_DRAW);
909 free (arrayOfVertices);
910
911 // normals buffer object array
912 const GLsizei sizeOfNormalsArray = (GLsizei) sizeof (arrayOfNormals[0]) * 3 * model.nVertices;
913 glGenBuffers (1, &model.normals);
914 glBindBuffer (GL_ARRAY_BUFFER, model.normals);
915 glBufferData (GL_ARRAY_BUFFER, sizeOfNormalsArray, arrayOfNormals, GL_STATIC_DRAW);
916 free (arrayOfNormals);
917
918 // texture coordinates buffer object array
919 const GLsizei sizeOfTextureCoordinateArray = (GLsizei) sizeof (arrayOfVertices[0]) * 2 * model.nVertices;
920 glGenBuffers (1, &model.tc);
921 glBindBuffer (GL_ARRAY_BUFFER, model.tc);
922 glBufferData (GL_ARRAY_BUFFER, sizeOfTextureCoordinateArray, arrayOfTextureCoordinates, GL_STATIC_DRAW);
923 free (arrayOfTextureCoordinates);
924
925 // unbind array buffer
926 glBindBuffer (GL_ARRAY_BUFFER, 0);
927
928 return model;
929}

931void associate_a_texture_to_model (struct model &m, const char *const path)
932{
933
934 static bool isFirstTimeBeingExecuted = true;
935 if (isFirstTimeBeingExecuted)
936 {
937 // DevIL setup - done once (slide 5) [class11]
938 ilInit ();
939 ilEnable (IL_ORIGIN_SET);
940 ilOriginFunc (IL_ORIGIN_LOWER_LEFT);
941 }
942
943 // for each image (slide 5) [class11]
944 ILuint image;
945 ilGenImages (1, &image);
946 ilBindImage (image);
947
948 const ILboolean has_loaded_successfully = ilLoadImage ((ILstring) path);
949 if (!has_loaded_successfully)
950 {
951 cerr << "[engine] failed loading texture file '" << path << "'"
952 << "\nERROR#" << ilGetError () << endl;
953 exit (EXIT_FAILURE);
954 }
955
956 // convert to RGBA (slide 6) [class11]
957 const ILboolean has_converted_image_sucessfully = ilConvertImage (IL_RGBA, IL_UNSIGNED_BYTE);
958 if (!has_converted_image_sucessfully)
959 {
960 cerr << "[engine] failed to convert texture '" << path << "'" << endl;
961 exit (EXIT_FAILURE);
962 }
963
964 // texture creation in OpenGL (slide 8) [class11]
965 // create a texture slot (slide 8) [class11]
966 glGenTextures (1, &m.tbo);
967
968 // bind the slot (slide 8) [class11]
969 glBindTexture (GL_TEXTURE_2D, m.tbo);
970
971 // define texture parameters (slide 8) [class11]
972 glTexParameteri (GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
973 glTexParameteri (GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
974
975 glTexParameteri (GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR_MIPMAP_LINEAR);
976 glTexParameteri (GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
977
978 // get the required info (slide 7) [class11]
979 const ILubyte *const texData = ilGetData ();
980 const GLsizei texture_width = ilGetInteger (IL_IMAGE_WIDTH);
981 const GLsizei texture_height = ilGetInteger (IL_IMAGE_HEIGHT);
982
983 // send texture data to OpenGL (slide 8) [class11]
984 glTexImage2D (GL_TEXTURE_2D, 0, GL_RGBA,
985 texture_width, texture_height, 0,
986 GL_RGBA, GL_UNSIGNED_BYTE, texData);
987
988 glGenerateMipmap (GL_TEXTURE_2D);
989
990 // unbind texture
991 glBindTexture (GL_TEXTURE_2D, 0);
992
993 isFirstTimeBeingExecuted = false;
994}

1031 // unbind array buffer
1032 glBindBuffer (GL_ARRAY_BUFFER, 0);
1033 // unbind texture (slide 10) [class11]
1034 glBindTexture (GL_TEXTURE_2D, 0);

1537void engine_run (int argc, char **argv)
1538{

1541 {

1543 exit (EXIT_FAILURE);

1545
1546

1548 glutInit (&argc, argv);
1549
1550 glutInitDisplayMode (GLUT_DEPTH | GLUT_DOUBLE | GLUT_RGBA);
1551 glutInitWindowPosition (100, 100);
1552 glutInitWindowSize (800, 800);
1553 glutCreateWindow ("engine");
1554
1555 // Required callback registry
1556 glutDisplayFunc (renderScene);
1557 loadProfile (profile[globalProfile]);
1558
1559
1560 // OpenGL settings
1561 glEnable (GL_DEPTH_TEST);
1562
1563 // activate 2D texturing (slide 10) [class11]
1564 glEnable (GL_TEXTURE_2D);
1565
1566 /*
1567 * For lighting to work properly when scales are applied to a model,
1568 * the following code below should be added to the initialization.
1569 * Activating this feature will result in normalizes normals after
1570 * applying the geometric transformations, and before applying lighting.
1571 */
1572 glEnable (GL_RESCALE_NORMAL);
1573
1574 // activate lighting (done once in initialization) (slides 5) [class9]
1575 glEnable (GL_LIGHTING);
1576 // glEnable (GL_LIGHTi) done when needed
1577
1578 // activate arrays (slide 12) [class11]
1579 glEnableClientState (GL_VERTEX_ARRAY);
1580 glEnableClientState (GL_NORMAL_ARRAY);
1581 glEnableClientState (GL_TEXTURE_COORD_ARRAY);

1583 /*
1584 * To allow for ambient colors to be reproduced without having
1585 * to activate the ambient component for all lights, the following
1586 * code should be added to the initialization:

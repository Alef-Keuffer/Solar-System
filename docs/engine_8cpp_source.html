59void spherical2Cartesian (double radius, double elevation, double azimuth, double *x, double *y, double *z)
60{
61
62 *x = radius * cos (elevation) * sin (azimuth);
63 *y = radius * sin (elevation);
64 *z = radius * cos (elevation) * cos (azimuth);
65}

67void cartesian2Spherical (double x, double y, double z, double *radius, double *azimuth, double *elevation)
68{
69 *radius = abs (sqrt (pow (x, 2) + pow (y, 2) + pow (z, 2)));
70 *elevation = asin (y / (*radius));
71 *azimuth = asin (x / (*radius * cos (*elevation)));
72
73}

255void changeSize (int w, int h)
256{
257
258 // Prevent a divide by zero, when window is too short
259 // (you cant make a window with zero width).

261 h = 1;
262

264 float ratio = (float) w * 1.0f / (float) h;
265
266 // Set the projection matrix as current
267 glMatrixMode (GL_PROJECTION);

269 glLoadIdentity ();

275 gluPerspective (globalFOV, ratio, globalNear, globalFar); //fox,near,far

297 DEFAULT_GLOBAL_CENTER_X = operations->at (i);
298 DEFAULT_GLOBAL_CENTER_Y = operations->at (i + 1);

301
302 DEFAULT_GLOBAL_UP_X = operations->at (i);
303 DEFAULT_GLOBAL_UP_Y = operations->at (i + 1);
304 DEFAULT_GLOBAL_UP_Z = operations->at (i + 2);
305 i += 3;
306
307 DEFAULT_GLOBAL_FOV = operations->at (i);
308 DEFAULT_GLOBAL_NEAR = operations->at (i + 1);
309 DEFAULT_GLOBAL_FAR = operations->at (i + 2);
310
311 unsigned int modelNo = 0;
312
313 for (i += 3; i < operations->size (); i++)
314 {
315 switch ((int) operations->at (i))
316 {
317 case ROTATE:
318 glRotatef (operations->at (i + 1),
319 operations->at (i + 2),
320 operations->at (i + 3),
321 operations->at (i + 4));
322 i += 4;
323 continue;
324 case TRANSLATE:
325 glTranslatef (operations->at (i + 1),
326 operations->at (i + 2),
327 operations->at (i + 3));
328 i += 3;
329 continue;
330 case SCALE:
331 glScalef (operations->at (i + 1),
332 operations->at (i + 2),
333 operations->at (i + 3));
334 i += 3;
335 continue;
336 case BEGIN_GROUP:glPushMatrix ();
337 continue;
338 case END_GROUP:glPopMatrix ();
339 continue;
340 case LOAD_MODEL:int stringSize = (int) operations->at (i + 1);
341 if (!hasPushedModels)
342 {
343 char modelName[stringSize + 1];
344
345 int j;
346 for (j = 0; j < stringSize; j++)
347 modelName[j] = (char) operations->at (i + 2 + j);
348
349 modelName[j] = 0;
350
351 globalModels.push_back (allocModel (modelName));
352 }
353 renderModel (globalModels.at (modelNo++));
354 i += stringSize + 1; //just to be explicit
355 continue;
356 }
357 }

448void keyboardFunc (unsigned char key, int xmouse, int ymouse)
449{
450 /*
451 Rotation:
452 x y z x↓ y↓ z↓
453 X Y Z x↑ y↑ z↑
454 angle:
455 , . θ-=s θ+=s
456 < > s↓ s↑
457 translation:
458 q w e y↓ z↑ y↑
459 a s d x↓ z↓ x↑
460 scaling:
461 u i o y↓ z↑ y↑
462 j k l x↓ z↓ x↑
463 background color: r g b R G B
464 camera:
465 1 2 3 ⎫ ! @ # ⎫ EyeX EyeY EyeZ
466 4 5 6 ⎬↓ $ % ^ ⎬↑ CenterX CenterY CenterZ
467 7 8 9 ⎭ & * ( ⎭ UpX UpY UpZ
468 */
469 switch (key)
470 {
471 /*Rotation*/
472 case 'x':globalRotateX -= 1;
473 break;
474
475 case 'X':globalRotateX += 1;
476 break;
477
478 case 'y':globalRotateY -= 1;
479 break;
480
481 case 'Y':globalRotateY += 1;
482 break;
483
484 case 'z':globalRotateZ -= 1;
485 break;
486
487 case 'Z':globalRotateZ += 1;
488 break;
489
490 case ',':globalAngle -= globalAngleStep;
491 break;
492
493 case '.':globalAngle += globalAngleStep;
494 break;
495
496 case '<':globalAngleStep /= 2;
497 break;
498
499 case '>':globalAngleStep *= 2;
500 break;
501
502 /*translation*/
503 case 'T':globalTranslateStep *= 2;
504 break;
505 case 't':
506 if (globalTranslateStep > 1)
507 globalTranslateStep /= 2;
508 break;
509
510 /*z-axis*/
511 case 'w':globalTranslateZ += globalTranslateStep;
512 break;
513 case 's':globalTranslateZ -= globalTranslateStep;
514 break;
515
516 /*x-axis*/
517 case 'a':globalTranslateX -= globalTranslateStep;
518 break;
519 case 'd':globalTranslateX += globalTranslateStep;
520 break;
521
522 /*y-axis*/
523 case 'q':globalTranslateY -= globalTranslateStep;
524 break;
525 case 'e':globalTranslateY += globalTranslateStep;
526 break;
527 /*end of translation*/
528
529 /*scaling*/
530
531 /*z-axis*/
532 case 'i':globalScaleZ += globalScaleStep;
533 break;
534 case 'k':globalScaleZ -= globalScaleStep;
535 break;
536
537 /*x-axis*/
538 case 'j':globalScaleX -= globalScaleStep;
539 break;
540 case 'l':globalScaleX += globalScaleStep;
541 break;
542
543 /*y-axis*/
544 case 'u':globalScaleY -= globalScaleStep;
545 break;
546 case 'o':globalScaleY += globalScaleStep;
547 break;
548 /*end of scaling*/
549
550 /*camera*/
551 case '`':globalEyeStep *= 2;
552 break;
553 case '~':
554 if (globalEyeStep > 1)
555 globalEyeStep /= 2;
556 break;
557
558 case '1':globalEyeX -= globalEyeStep;
559 break;
560 case '!':globalEyeX += globalEyeStep;
561 break;
562 case '2':globalEyeY -= globalEyeStep;
563 break;
564 case '@':globalEyeY += globalEyeStep;
565 break;
566 case '3':globalEyeZ -= globalEyeStep;
567 break;
568 case '#':globalEyeZ += globalEyeStep;
569 break;
570
571 case '4':globalCenterX -= globalCenterStep;
572 break;
573 case '$':globalCenterX += globalCenterStep;
574 break;
575 case '5':globalCenterY -= globalCenterStep;
576 break;
577 case '%':globalCenterY += globalCenterStep;
578 break;
579 case '6':globalCenterZ -= globalCenterStep;
580 break;
581 case '^':globalCenterZ += globalCenterStep;
582 break;
583 case '7':globalUpX -= globalUpStep;
584 break;
585 case '&':globalUpX += globalUpStep;
586 break;
587 case '8':globalUpY -= globalUpStep;
588 break;
589 case '*':globalUpY += globalUpStep;
590 break;
591 case '9':globalUpZ -= globalUpStep;
592 break;
593 case '(':globalUpZ += globalUpStep;
594 break;
595
596 /*reset environment*/
597 case '0':env_load_defaults ();
598 break;
599
600 default:break;
601 }
602 redisplay (); //request display() call ASAP
603}

608void mouseFunc (int button, int state, int x, int y)
609{
610 // Wheel reports as button 3(scroll up) and button 4(scroll down)
611 if ((button == 3) || (button == 4)) // It's a wheel event
612 {
613 // Each wheel event reports like a button click, GLUT_DOWN then GLUT_UP
614 if (state == GLUT_UP)
615 return; // Disregard redundant GLUT_UP events
616
617 if (button == 3)
618 {
619 globalRadius *= .8f;
620 if (globalRadius < 1.0f)
621 globalRadius = 1.0f;
622 }
623 else
624 globalRadius *= 1.5;
625 printf ("Scroll %s At %d %d\n", (button == 3) ? "Up" : "Down", x, y);
626 }
627 else
628 { // normal button event
629 printf ("Button %s At %d %d\n", (state == GLUT_DOWN) ? "Down" : "Up", x, y);
630 if (button == GLUT_LEFT_BUTTON)
631 {
632 if (state == GLUT_DOWN)
633 {
634 globalMouseLeftButton = true;
635 globalXPrev = x;
636 globalYPrev = y;
637 }
638 if (state == GLUT_UP)
639 {
640 globalMouseLeftButton = false;
641 }
642 }
643 else if (button == GLUT_RIGHT_BUTTON)
644 {
645 if (state == GLUT_DOWN)
646 {
647 int window_width = glutGet (GLUT_WINDOW_WIDTH);
648 int window_height = glutGet (GLUT_WINDOW_HEIGHT);
649
650 GLbyte color[4];
651 GLfloat depth;
652 GLuint index;
653
654 glReadPixels (x, window_height - y - 1, 1, 1, GL_RGBA, GL_UNSIGNED_BYTE, color);
655 glReadPixels (x, window_height - y - 1, 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &depth);
656 glReadPixels (x, window_height - y - 1, 1, 1, GL_STENCIL_INDEX, GL_UNSIGNED_INT, &index);
657
658 double ox, oy, oz;
659
660 double model[16];
661 glGetDoublev (GL_MODELVIEW_MATRIX, model);
662 double proj[16];
663 glGetDoublev (GL_PROJECTION_MATRIX, proj);
664 int view[4];
665 glGetIntegerv (GL_VIEWPORT, view);
666 gluUnProject (x, window_height - y - 1, depth,
667 model,
668 proj,
669 view,
670 &ox, &oy, &oz);
671
672 cartesian2Spherical (ox, globalEyeY, oz,
673 &globalRadius, &globalAzimuth, &globalElevation);
674
675 globalCenterX = ox;
676 globalCenterZ = oz;
677
678 fprintf (stderr, "%f %f %f", ox, oy, oz);
679 }
680 }
681 }
682
683 redisplay ();
684}

686void motionFunc (int x, int y)
687{
688 if (globalMouseLeftButton)
689 {
690 const double stepAlfa = 0.035;
691 const double stepBeta = 0.035;
692
693 if (x < globalXPrev)
694 globalAzimuth += stepAlfa;
695 else if (x > globalXPrev)
696 globalAzimuth -= stepAlfa;
697
698 if (y < globalYPrev)
699 {
700 globalElevation -= stepBeta;
701 if (globalElevation < -1.5)
702 globalElevation = -1.5;
703 }
704 else if (y > globalYPrev)
705 {
706 globalElevation += stepBeta;
707 if (globalElevation > 1.5)
708 globalElevation = 1.5;
709 }
710
711 globalXPrev = x;
712 globalYPrev = y;
713 }
714 redisplay ();
715}
